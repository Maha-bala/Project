void main()
{
  int a=85;
  if(a>=90)
  {
    print("I grade");
  }
  else if(a>=80 && a<=89)
  {
    print("II grade");
  }
  else if(a>=70 && a<=79)
  {
    print("III grade");
  }
  else if(a>=60 && a<=69)
  {
    print("IV grade");
  }
  else
  {
    print("V grade");
  }
}
